Database Type:
	- RDBMS
	- Hard Delete

1. User {
	id Int autoincrement
	email String(50) unique
	name String(50)
	image ? String(255)
	createdAt DateTime
	updatedAt DateTime
}

2. Post {
	id Int autoincrement
	author Int referenceTo(Table: User, Columns: id)
	kategori JSON.stringify(String(100))
	title String(100)
	content Text()
	image ? String(255)
	// like Int optional
	createdAt DateTime
	updatedAt DateTime
	// deleted Boolean
}

3. Forum {
	id Int autoincrement
	author Int referenceTo(Table: User, Columns: id)
	kategori JSON.stringify(String(100))
	title String(100)
	// like Int optional
	createdAt DateTime
	updatedAt DateTime
}

4. Komentar_Forum {
	id Int autoincrement
	author Int referenceTo(Table: User, Columns: id)
	forumId Int referenceTo(Table: Forum, Columns: id)
	content Text()
	// like Int optional
	createdAt DateTime
	updatedAt DateTime
}

5. Komentar_Post {
	id Int autoincrement
	author Int referenceTo(Table: User, Columns: id)
	postId Int referenceTo(Table: Post, Columns: id)
	content Text()
	// like Int optional
	createdAt DateTime
	updatedAt DateTime
}

6. Log {
	action String(100)
	message String(100)
	status Int
}
